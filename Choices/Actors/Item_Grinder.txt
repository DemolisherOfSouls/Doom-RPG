Actor Item_Grinder : Item_Weapon
{
	States
	{
	Spawn:
		GGdr A 15
		GGdr A 5 Bright
		Loop
	Pickup:
		TNT1 A 0 A_JumpIf(!CallACS("WeaponPickup", 5, W_GRINDER), "Pickup.Fail")
		TNT1 A 0 A_GiveInventory("Ammo_Clip", 40)
		Goto Pickup.Force
	}
}

Actor Item_GrinderDropped : Item_Grinder
{
	States
	{
	Pickup:
		TNT1 A 0 A_JumpIf(!CallACS("WeaponPickup", 5, W_GRINDER), "Pickup.Fail")
		Goto Pickup.Force
	}
}
