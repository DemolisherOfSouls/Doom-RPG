Actor Item_Chaingun : Item_Weapon replaces Chaingun
{
	SpawnID 28
	
	States
	{
	Spawn:
		MGUN A 15
		MGUN A 5 Bright
		Loop
	Pickup:
		TNT1 A 0 A_JumpIf(!CallACS("WeaponPickup", 5, W_CHAINGUN), "Pickup.Fail")
		TNT1 A 0 A_GiveInventory("Ammo_Clip", 60)
		Goto Pickup.Force
	}
}

Actor Item_ChaingunDropped : Item_Chaingun
{
	States
	{
	Pickup:
		TNT1 A 0 A_JumpIf(!CallACS("WeaponPickup", 5, W_CHAINGUN), "Pickup.Fail")
		Goto Pickup.Force
	}
}
