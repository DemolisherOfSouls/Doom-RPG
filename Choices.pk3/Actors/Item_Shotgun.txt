Actor Item_Shotgun : Item_Weapon replaces Shotgun
{
	SpawnID 27
	
	States
	{
	Spawn:
		SHOT A 15
		SHOT A 5 Bright
		Loop
	Pickup:
		TNT1 A 0 A_JumpIf(!CallACS("WeaponPickup", 3, W_SHOTGUN), "Pickup.Fail")
		TNT1 A 0 A_GiveInventory("AmmoMag_Shotgun", 6)
		Goto Pickup.Force
	}
}

Actor Item_ShotgunDropped : Item_Shotgun
{
	States
	{
	Pickup:
		TNT1 A 0 A_JumpIf(!CallACS("WeaponPickup", 3, W_SHOTGUN), "Pickup.Fail")
		Goto Pickup.Force
	}
}
