Actor WeaponProj_TracerTrail
{
	Height 4
	Radius 2
	+NOGRAVITY
	+NOINTERACTION
	
	var int user_offset;
	
	States
	{
	Spawn:
		BULL B 0
		Goto SpawnLoop
	SpawnLoop:
		BULL B 0 A_JumpIf(user_offset == -56, "Fade")
		BULL B 0 A_SpawnItemEx("WeaponProj_TracerTrail2", user_offset, 0, 0, velx, vely, velz, 0, SXF_NoCheckPosition)
		BULL B 0 A_SetUserVar("user_offset", user_offset - 7)
		Loop
	Fade:
		BULL B 1 A_FadeOut(0.05, true)
		Wait
	}
}

Actor WeaponProj_TracerTrail2 : WeaponProj_TracerTrail
{
	States
	{
	Spawn:
		BULL B 0
		Goto Fade
	}
}
