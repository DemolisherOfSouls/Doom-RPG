Actor WeaponProj_Bullet : WeaponProj_Base
{
	Height 4
	Radius 2
	Speed 64
	+BLOODSPLATTER
	+SEEKERMISSILE
	+BRIGHT
	
	States
	{
	Spawn:
		BULL A 0 NoDelay A_JumpIf(CallACS("Skill_BloodyMess"), "SetGibs")
		BULL A 0 A_Jump(Always, "Select")
	Select:
		BULL A 0 A_Jump(Always, "Normal")
	SetGibs:
		BULL A 0 A_ChangeFlag("EXTREMEDEATH", true)
		BULL A 0 A_Jump(Always, "Select")
	Normal:
		BULL A -1 A_JumpIf(CallACS("Skill_HomingRounds"), "Normal.Homing")
	Normal.Homing:
		BULL A 1 A_SeekerMissile(1, 2, SMF_Look | SMF_Precise, 128, 5)
		Loop
	Tracer:
		BULL B 0 A_JumpIf(CallACS("Skill_HomingRounds"), "Tracer.Homing")
		BULL B 1 A_SpawnItemEx("WeaponProj_TracerTrail", 0, 0, 0, velx / 64, vely / 64, velz / 64, 0, SXF_NoCheckPosition)
		Wait
	Tracer.Homing:
		BULL B 0 A_SeekerMissile(1, 2, SMF_Look | SMF_Precise, 128, 5)
		BULL B 1 A_SpawnItemEx("WeaponProj_TracerTrail", 0, 0, 0, velx / 64, vely / 64, velz / 64, 0, SXF_NoCheckPosition)
		Loop
	Death:
		TNT1 A 1 A_SpawnItem("BulletPuff")
		Stop
	XDeath:
		TNT1 A 1
		Stop
	}
}
